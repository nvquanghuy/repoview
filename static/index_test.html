// Test: keyboard shortcut dispatch behavior
(function testShortcutDispatch() {
  function isTypingTarget(target) {
    var tag = target && target.tagName;
    return tag === 'INPUT' || tag === 'TEXTAREA';
  }

  function dispatchShortcut(e, state) {
    if (e.metaKey || e.ctrlKey || e.altKey) return false;
    if (isTypingTarget(e.target) && e.key !== 'Escape') return false;

    if (e.key === 'Escape') {
      if (state.searchOpen) {
        e.preventDefault();
        state.searchOpen = false;
        return true;
      }
      if (state.shortcutsOpen) {
        e.preventDefault();
        state.shortcutsOpen = false;
        return true;
      }
      if (state.outlineOpen) {
        e.preventDefault();
        state.outlineOpen = false;
        return true;
      }
      return false;
    }

    if ((e.key === 't' || e.key === '/') && !state.searchOpen && !state.shortcutsOpen) {
      e.preventDefault();
      state.searchOpen = true;
      return true;
    }
    if (e.key === '?' && !state.searchOpen && !state.shortcutsOpen) {
      e.preventDefault();
      state.shortcutsOpen = true;
      return true;
    }
    return false;
  }

  var state = { searchOpen: false, shortcutsOpen: false, outlineOpen: false };
  var prevented = false;
  dispatchShortcut({
    target: { tagName: 'DIV' },
    key: '/',
    preventDefault: function() { prevented = true; }
  }, state);
  assertEqual(prevented, true, "Slash shortcut opens search");
  assertEqual(state.searchOpen, true, "Slash shortcut sets searchOpen");

  state = { searchOpen: false, shortcutsOpen: false, outlineOpen: false };
  prevented = false;
  dispatchShortcut({
    target: { tagName: 'INPUT' },
    key: 't',
    preventDefault: function() { prevented = true; }
  }, state);
  assertEqual(prevented, false, "Typing target ignores non-Escape shortcuts");
  assertEqual(state.searchOpen, false, "Typing target does not open search");

  state = { searchOpen: false, shortcutsOpen: false, outlineOpen: false };
  prevented = false;
  dispatchShortcut({
    target: { tagName: 'DIV' },
    key: '?',
    preventDefault: function() { prevented = true; }
  }, state);
  assertEqual(prevented, true, "Question mark opens shortcuts");
  assertEqual(state.shortcutsOpen, true, "Question mark sets shortcutsOpen");

  state = { searchOpen: true, shortcutsOpen: true, outlineOpen: true };
  prevented = false;
  dispatchShortcut({
    target: { tagName: 'DIV' },
    key: 'Escape',
    preventDefault: function() { prevented = true; }
  }, state);
  assertEqual(prevented, true, "Escape prevents default when closing panels");
  assertEqual(state.searchOpen, false, "Escape closes search first");
  assertEqual(state.shortcutsOpen, true, "Escape keeps shortcuts open when search was open");
})();

// Test: live API call to /api/file (only runs when served by repoview)
(async function testLiveAPI() {
  try {
    var res = await fetch('/api/file?path=testdata/hello.md');
    if (res.ok) {
      var data = await res.json();
      assertEqual(typeof data.content, 'string', 'Live API: content is string');
      assertEqual(data.isMarkdown, true, 'Live API: isMarkdown is true');
      assertEqual(data.name, 'hello.md', 'Live API: name is hello.md');
      assertContains(data.content, '
', 'Live API: content contains rendered HTML');
    } else {
      console.log('Skipping live API test (file not found — not running from repo root?)');
    }
  } catch (e) {
    console.log('Skipping live API tests (not served by repoview)');
  }
  renderResults();
})();
